import numpy as np  # not needed in example below

def poly3(x):
    """ POLY3: Evaluates the cubic polynomial y = a0 + a1*x+a2*x**2+a3*x**3
    using Horner's method, where a0=1, a1=-1, a2=2, a3=1.
    """
    
    a0 = 1
    a1 = -1
    a2 = 2
    a3 = 1
    
    y = a0 + x*(a1 + x*(a2+x*a3))
    
    return y

myx = -2
new = poly3(myx)
